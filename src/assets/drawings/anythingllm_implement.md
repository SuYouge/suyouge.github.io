// 聊天端点
router.post('/chat', async (req, res) => {
  const { message, workspaceId } = req.body;
  const workspace = await Workspace.get(workspaceId);
  
  // 1. 检索相关文档
  const context = await VectorDb.similarityResponse(
    workspace,
    message
  );
  
  // 2. 调用 LLM
  const llm = getLLMProvider();
  const response = await llm.sendChat(
    formatPrompt(context, message)
  );
  
  // 3. 流式返回
  res.writeHead(200, {
    'Content-Type': 'text/event-stream'
  });
  response.pipe(res);
}); ^9YRlvEHz
